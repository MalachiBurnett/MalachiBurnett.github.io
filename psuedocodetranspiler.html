        def transpile_to_python(pseudocode):
            # A simple example of pseudocode to Python transpilation.
            python_code = pseudocode.replace("OUTPUT ", "print(").replace(")", ")"
            python_code = python_code.replace("INPUT ", "input("))
            return python_code

        def run_transpiler(pseudocode):
            python_code = transpile_to_python(pseudocode)
            try:
                # Execute the Python code and capture its output.
                exec_result = eval(python_code)
            except Exception as e:
                exec_result = f"Error: {e}"
            return python_code, exec_result
